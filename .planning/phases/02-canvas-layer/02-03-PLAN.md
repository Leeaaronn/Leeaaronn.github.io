**Implementation of updateGlobe:**

```javascript
export function updateGlobe({ activeSection, progress }) {
  if (!renderer) return; // guard if not initialized
  
  const canvas = renderer.domElement;
  
  if (activeSection === 'hero') {
    // GLOB-05: Centered, full opacity, slowly spinning (default state)
    camera.position.z = CAMERA_Z_DEFAULT; // 5
    canvas.style.opacity = '1';
    globeGroup.position.set(0, 0, 0);
    
  } else if (activeSection === 'about') {
    // GLOB-06: Zoom in toward LA, rotate globe to center LA
    // Interpolate camera z from 5 to 3.2 based on progress
    camera.position.z = CAMERA_Z_DEFAULT - (progress * 1.8);
    canvas.style.opacity = '1';
    
    // Rotate globe to show LA — target rotation where LA faces camera
    // LA is at lon -118.25, which means we need to rotate globe to bring that longitude forward
    // Target Y rotation: convert LA longitude to radians: (-118.25 + 90) * PI/180 ~ -0.49 rad
    // But the globe is continuously rotating, so we lerp toward a target
    // Instead: reduce rotation speed and shift globe position slightly
    // Shift globe slightly left and up to "zoom to LA" effect
    const targetX = -0.3 * progress;
    const targetY = 0.2 * progress;
    globeGroup.position.x += (targetX - globeGroup.position.x) * 0.05;
    globeGroup.position.y += (targetY - globeGroup.position.y) * 0.05;
    
  } else {
    // GLOB-07: Past about — fade out globe, stars remain
    // Quick fade: opacity goes from 1 to 0 over the first 30% of the next section
    const fadeProgress = Math.min(progress / 0.3, 1);
    canvas.style.opacity = String(1 - fadeProgress);
    
    // Reset position for when user scrolls back
    if (parseFloat(canvas.style.opacity) <= 0.01) {
      canvas.style.opacity = '0';
    }
  }
}
```

**Also add a `CAMERA_Z_DEFAULT` constant** at module scope (value 5) if not already there from Plan 02 action (it should be, but verify).

**Key behavior notes:**
- The globe's continuous Y rotation in the animation loop (0.002/frame) keeps running in all states
- In the 'about' state, the zoom + position shift creates the "zoom to LA" effect
- In 'past about' states, the canvas opacity fades to 0 but the stars canvas (separate renderer at z-index 0) stays at opacity 1
- The globe position resets smoothly when scrolling back up to hero thanks to the lerp

```javascript
import './style.css'
import { initGlobe, updateGlobe } from './globe.js'
import { initStars } from './stars.js'
import { initScroll } from './scroll.js'

document.addEventListener('DOMContentLoaded', () => {
  // Phase 2: Canvas layers
  initStars();
  initGlobe();
  
  // Scroll-driven globe behavior
  // Body is the scroll container (scroll-snap-type y mandatory)
  const sections = Array.from(document.querySelectorAll('.section'));
  
  function getScrollState() {
    const scrollTop = document.body.scrollTop || document.documentElement.scrollTop;
    const viewportHeight = window.innerHeight;
    
    // Find which section is most visible
    let activeSection = 'hero';
    let progress = 0;
    
    for (const section of sections) {
      const rect = section.getBoundingClientRect();
      // Section is "active" when its top is within half a viewport of the top
      if (rect.top <= viewportHeight * 0.5 && rect.bottom > viewportHeight * 0.5) {
        activeSection = section.id;
        // Progress: 0 when section top is at viewport top, 1 when section bottom is at viewport top
        progress = Math.max(0, Math.min(1, -rect.top / viewportHeight));
        break;
      }
    }
    
    return { activeSection, progress };
  }
  
  // Use body scroll since body is the snap container
  document.body.addEventListener('scroll', () => {
    const state = getScrollState();
    updateGlobe(state);
  }, { passive: true });
  
  // Also fire once on load to set initial state
  updateGlobe(getScrollState());
  
  // Phase 3: scroll UI (still a stub)
  // initScroll();
});
```

- `{ passive: true }` on scroll listener for performance
- `getScrollState()` uses `getBoundingClientRect()` to determine which section is active and how far through it the user has scrolled
- Call `updateGlobe(getScrollState())` once on load to set the initial hero state
